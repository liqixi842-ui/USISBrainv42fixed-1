我要把 /report 参数解析升级为「D 模式」：同时支持 3 种写法：

1）/report NVDA brand=VADA firm=Aberdeen_Investments analyst=Anthony_Venn_Dutton
2）/report NVDA brand="VADA" firm="Aberdeen Investments" analyst="Anthony Venn Dutton"
3）/report NVDA brand=VADA firm=Aberdeen Investments analyst=Anthony Venn Dutton

要求：最终都解析为：
- brand = "VADA"
- firm = "Aberdeen Investments"
- analyst = "Anthony Venn Dutton"

请在 v3_dev/services/devBotHandler.js 中，对 /report 文本做以下处理：

【1】获取原始文本（去掉 /report 和 symbol 部分）

例如：
- 原始 text: "/report NVDA brand=VADA firm=Aberdeen Investments analyst=Anthony Venn Dutton"
- 解析出 symbol="NVDA"，剩余 paramString="brand=VADA firm=Aberdeen Investments analyst=Anthony Venn Dutton"

【2】实现一个鲁棒解析函数 parseParams(paramString)

伪代码逻辑：

function parseParams(str) {
  const params = {};
  let currentKey = null;
  let currentValue = [];

  // 先按空格拆分
  const tokens = str.trim().split(/\s+/);

  for (const token of tokens) {
    if (token.includes('=')) {
      // 遇到新的 key=value，先把前一个 key 写入
      if (currentKey) {
        params[currentKey] = currentValue.join(' ').trim();
      }

      const [rawKey, rawValue] = token.split('=');
      currentKey = rawKey.trim().toLowerCase();
      currentValue = [rawValue];

    } else if (currentKey) {
      // 没有 '=', 说明是上一个 key 的续写部分
      currentValue.push(token);
    }
  }

  // 最后一个 key
  if (currentKey) {
    params[currentKey] = currentValue.join(' ').trim();
  }

  // 处理引号和下划线
  for (const key of Object.keys(params)) {
    let v = params[key];

    // 去掉首尾引号
    v = v.replace(/^"(.*)"$/,'$1').replace(/^'(.*)'$/,'$1');

    // 把下划线还原成空格
    v = v.replace(/_/g, ' ');

    params[key] = v.trim();
  }

  return params;
}
